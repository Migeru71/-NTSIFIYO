@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-017: Eliminar actividades

actor "Docente / Admin" as U
boundary "PanelActividadesView" as View
control "ApiService" as Api
control "ActivityController" as Ctrl
entity "ActivityService" as Srv
entity "ActivityRepository" as Repo
database "MySQL" as DB

autonumber

U -> View: Ingresa al panel de actividades
View -> Api: fetchActivities()
Api -> Ctrl: GET /api/activities
Ctrl -> Srv: listActivities()
Srv -> Repo: findAll()
Repo -> DB: SELECT * FROM activities
DB --> Repo: activityList
Repo --> Srv: activityList
Srv --> Ctrl: activityList
Ctrl --> Api: 200 OK
Api --> View: RenderList

U -> View: Selecciona una actividad y opción "Eliminar"
View -> Api: deleteActivity(id)
Api -> Ctrl: DELETE /api/activities/{id}
Ctrl -> Srv: processDeletion(id)

group Validaciones de Seguridad y Estado (Backend)
    alt #FFFDD0 La actividad no fue creada por ese docente
        Srv -> Srv: checkOwnership(id, userId)
        note right: Los Administradores pueden\neliminar cualquier actividad
        Srv --> Ctrl: throw UnauthorizedException
        Ctrl --> Api: 403 Forbidden
        Api --> View: Error detectado
        View --> U: Mostrar: "La actividad no ha sido creada por usted"
    end

    alt #FFFDD0 La actividad tiene instancias activas
        Srv -> Srv: checkActiveAssignments(id)
        note right: Se verifica si hay grupos con esta\nactividad asignada actualmente
        Srv --> Ctrl: throw ActiveInstanceException
        Ctrl --> Api: 409 Conflict
        Api --> View: Error detectado
        View --> U: Mostrar: "No se puede eliminar: tiene instancias activas"
    end
end

Srv -> Repo: softDelete(id)
note right: Se mantiene el registro histórico de\nhaber realizado la actividad

Repo -> DB: UPDATE activities SET deleted_at = NOW() WHERE id = ?
DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: success
Ctrl --> Api: 200 OK
Api --> View: notifySuccess
View --> U: Notifica eliminación exitosa del catálogo

@enduml
