title CU-008: Gestionar experiencia y niveles de usuarios

actor "Usuario" as U
boundary "ActividadView" as View
control "ApiService" as Api
control "ExperienceController" as Ctrl
entity "ExperienceService" as Srv
entity "UserRepository" as Repo
database "MySQL" as DB

autonumber

U -> View: Completa una actividad
View -> Api: notifyActivityCompletion(activityId, results)
Api -> Ctrl: POST /api/experience/process-reward (dto)
Ctrl -> Srv: processActivityReward(dto)

Srv -> Repo: findUserById(userId)
Repo -> DB: SELECT total_experience, level FROM user WHERE id = ?
DB --> Repo: UserEntity
Repo --> Srv: UserEntity

group Lógica de Gamificación
    Srv -> Srv: calculateGainedXp(results, activityId)
    note right: La experiencia puede estar definida\npor el sistema o por el docente
    
    Srv -> Srv: checkLevelUp(currentExp, gainedExp)
    note right: Determina si el usuario alcanza un nuevo nivel
end

Srv -> Repo: updateProgress(updatedUser)
Repo -> DB: UPDATE user SET total_experience = ?, level = ? WHERE id = ?
DB --> Repo: success
Repo --> Srv: confirmation

Srv --> Ctrl: RewardResponseDto (xpGained, currentLevel, isLevelUp)
Ctrl --> Api: 200 OK (RewardData)
Api --> View: showExperienceIncrement(RewardData)
View --> U: Retorna vista con incremento de experiencia y nivel

@enduml
