@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-004: Asignar alumnos y docentes a grupos

actor "Administrador" as Admin
boundary "GestionGruposView" as View
control "ApiService" as Api
control "GroupController" as Ctrl
entity "GroupService" as Srv
entity "UserRepository" as UserRepo
entity "GroupRepository" as Repo
database "MySQL" as DB

autonumber

Admin -> View: Accede a vista de grupos (1ro a 6to)
View -> Api: fetchGroups()
Api -> Ctrl: GET /api/groups
Ctrl -> Srv: getAllGroups()
Srv -> Repo: findAll()
Repo -> DB: SELECT * FROM group
DB --> Repo: groupList
Repo --> Srv: groupList
Srv --> Ctrl: groupList
Ctrl --> Api: 200 OK
Api --> View: RenderGroups

Admin -> View: Selecciona un grupo y opción "Agregar"
View -> Api: fetchAvailableUsers()
Api -> Ctrl: GET /api/users/available

    Ctrl -> Srv: getAvailableUsers()
    Srv -> UserRepo: findUnassignedStudentsAndAllTeachers()
    UserRepo -> DB: SELECT * FROM user WHERE id NOT IN (...)
    DB --> UserRepo: candidatesList
    UserRepo --> Srv: candidatesList
    
    alt #FFFDD0 No hay más alumnos o docentes registrados para agregar
        Srv --> Ctrl: throw NoUsersAvailableException
        Ctrl --> Api: 404 Not Found ("Sin candidatos")
        Api --> View: Error detectado
        View --> Admin: Mostrar: "No hay más alumnos o docentes para agregar"
    end

Srv --> Ctrl: candidatesList
Ctrl --> Api: 200 OK (List)
Api --> View: ShowCandidateSelection

Admin -> View: Selecciona usuario y confirma asignación
View -> Api: assignUser(groupId, userId)
Api -> Ctrl: POST /api/groups/assign (dto)
Ctrl -> Srv: assignUserToGroup(dto)

alt #FFFDD0 Alumno ya registrado en otro grupo
    Srv -> Srv: validateExclusivity(userId)
    note right: Validación de seguridad adicional
    Srv --> Ctrl: throw AlreadyAssignedException
    Ctrl --> Api: 409 Conflict
    Api --> View: Error detectado
    View --> Admin: Mostrar: "El alumno seleccionado ya está en otro grupo"
end

Srv -> Repo: saveAssignment(groupId, userId)
Repo -> DB: INSERT INTO student/teacher_group
DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: SuccessDto
Ctrl --> Api: 200 OK
Api --> View: SuccessNotification
View --> Admin: Notifica asignación exitosa

@enduml
