@startuml

    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-003: Eliminar usuarios

actor "Administrador" as Admin
boundary "ListadoUsuariosView" as View
control "ApiService" as Api
control "UserController" as Ctrl
entity "UserService" as Srv
entity "UserRepository" as Repo
database "MySQL" as DB

autonumber

Admin -> View: Accede a vista de listado
View -> Api: fetchAllUsers()
Api -> Ctrl: GET /api/users
Ctrl -> Srv: getUsersList()
Srv -> Repo: findAll()
Repo -> DB: SELECT * FROM users
DB --> Repo: data / empty
Repo --> Srv: UserList

alt #FFFDD0 No hay usuarios registrados
    Srv --> Ctrl: throw NoDataException
    Ctrl --> Api: 404 Not Found (Empty)
    Api --> View: Muestra estado vacío
    View --> Admin: Solicitar registrar usuario primero
end

Srv --> Ctrl: UserList
Ctrl --> Api: 200 OK (List)
Api --> View: RenderList
View --> Admin: Muestra listado seccionado

Admin -> View: Selecciona usuario y confirma eliminación
View -> Api: deleteUser(id)
Api -> Ctrl: DELETE /api/users/{id}?ut
Ctrl -> Srv: processUserRemoval(id)

Srv -> Repo: deleteById(id)
Repo -> DB: DELETE FROM user WHERE id = ?
note right: Se mantienen registros de actividades\npara fines estadísticos

DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: confirmation
Ctrl --> Api: 200 OK
Api --> View: SuccessNotification
View --> Admin: Notifica eliminación exitosa

@enduml
