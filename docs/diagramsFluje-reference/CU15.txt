@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-015: Gestionar alumnos dentro de un grupo

actor "Administrador" as Admin
boundary "GestionGruposView" as View
control "ApiService" as Api
control "GroupController" as Ctrl
entity "GroupService" as Srv
entity "UserRepository" as Repo
database "MySQL" as DB

autonumber

Admin -> View: Accede al panel de grupos
View -> Api: fetchAllGroups()
Api -> Ctrl: GET /api/groups
Ctrl -> Srv: getGroupsList()
Srv -> Repo: findAllGroups()
Repo -> DB: SELECT * FROM group
DB --> Repo: groupList
Repo --> Srv: groupList
Srv --> Ctrl: groupList
Ctrl --> Api: 200 OK
Api --> View: RenderGroups

Admin -> View: Selecciona un grupo específico
View -> Api: fetchGroupMembers(groupId)
Api -> Ctrl: GET /api/groups/{id}/members
Ctrl -> Srv: getStudentsInGroup(id)
Srv -> Repo: findStudentsByGroupId(id)
Repo -> DB: SELECT * FROM user WHERE group_id = ?
DB --> Repo: studentList
Repo --> Srv: studentList
Srv --> Ctrl: studentList
Ctrl --> Api: 200 OK
Api --> View: Muestra opciones de gestión (Mover / Sacar)

Admin -> View: Selecciona alumno y opción (Mover o Sacar)
note right: Si es "Mover", selecciona el nuevo grupo

View -> Api: updateStudentEnrollment(dto)
Api -> Ctrl: PATCH /api/groups/enrollment (dto)
Ctrl -> Srv: reassignStudent(dto)

group Validación de Reglas de Negocio (Backend)
    alt #FFFDD0 El grupo destino es el mismo que el actual
        Srv --> Ctrl: throw RedundantAssignmentException
        Ctrl --> Api: 400 Bad Request
        Api --> View: Error detectado
        View --> Admin: Mostrar: "El alumno ya se encuentra en ese grupo"
    end
end

Srv -> Repo: updateGroupReference(studentId, newGroupId)
note right: "Sacar" del grupo equivale a\nsetear group_id como NULL

Repo -> DB: UPDATE user SET group_id = ? WHERE id = ?
DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: success
Ctrl --> Api: 200 OK
Api --> View: notifyUpdateSuccess
View --> Admin: Notifica actualización correcta del estado del alumno

@enduml
