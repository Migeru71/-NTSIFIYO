boundary "DictionaryView" as View

control "DictionaryController" as Ctrl
entity "DictionaryService" as Srv
entity "DictionaryRepository" as Repo
database "MySQL" as DB

autonumber

Admin -> View: Accede al panel de diccionario
View -> Api: fetchDictionaryItems()
Api -> Ctrl: GET /api/dictionary
Ctrl -> Srv: getAllItems()
Srv -> Repo: findAll()
Repo -> DB: SELECT * FROM dictionary
DB --> Repo: itemsList
Repo --> Srv: itemsList
Srv --> Ctrl: itemsList
Ctrl --> Api: 200 OK
Api --> View: RenderDictionary

Admin -> View: Selecciona palabra o frase y solicita eliminar
View -> Admin: Solicita confirmación de borrado

Admin -> View: Confirma eliminación
View -> Api: deleteDictionaryItem(id)
Api -> Ctrl: DELETE /api/dictionary/{id}
Ctrl -> Srv: removeElement(id)

group Validación de Integridad (Backend)
    Srv -> Repo: existsById(id)
    Repo -> DB: SELECT count(*) FROM dictionary WHERE id = ?
    DB --> Repo: count
    
    alt #FFFDD0 El elemento ya no existe o fue eliminado previamente
        Repo --> Srv: false
        Srv --> Ctrl: throw ResourceNotFoundException
        Ctrl --> Api: 404 Not Found
        Api --> View: Error detectado
        View --> Admin: Mostrar: "El elemento ya no se encuentra en el diccionario"
    end
end

Srv -> Repo: deleteById(id)
Repo -> DB: DELETE FROM dictionary WHERE id = ?

DB --> Repo: success

Srv --> Ctrl: success
