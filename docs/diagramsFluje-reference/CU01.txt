@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-001: Inicio de Sesión

actor "Usuario" as U
boundary "LoginView" as View
control "ApiService" as Api
control "AuthController" as Ctrl
entity "AuthService" as Srv
entity "UserRepository" as Repo
database "MySQL" as DB

autonumber

U -> View: Ingresa identificador y selecciona tipo
View -> Api: login(credentials)
Api -> Ctrl: POST /api/auth/login (dto)
Ctrl -> Srv: authenticate(dto)

Srv -> Repo: findByUsernameAndType(username, type)
Repo -> DB: SELECT * FROM user WHERE ...
DB --> Repo: data / null
Repo --> Srv: UserEntity / null

alt #FFFDD0 Usuario no encontrado
    Srv --> Ctrl: throw UserNotFoundException
    Ctrl --> Api: 404 Not Found ("Usuario no encontrado")
    Api --> View: Error detectado
    View --> U: Mostrar: "Usuario no encontrado"
end

Srv -> Srv: validateHash(password, storedHash)

alt #FFFDD0 Contraseña incorrecta
    Srv --> Ctrl: throw InvalidCredentialsException
    Ctrl --> Api: 401 Unauthorized ("Credenciales inválidas")
    Api --> View: Error detectado
    View --> U: Mostrar: "Credenciales inválidas"
end

Srv -> Srv: generateJWT(UserEntity)
note right: Genera token de seguridad

Srv --> Ctrl: JwtResponse
Ctrl --> Api: 200 OK (JWT)

Api --> View: AuthSuccess
View -> View: Guardar JWT localmente
View -> U: Redirigir al panel principal

@enduml
