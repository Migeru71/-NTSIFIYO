@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-010: Visualizar diccionario

actor "Usuario" as U
boundary "DictionaryView" as View
control "ApiService" as Api
control "WordsController" as Ctrl
entity "WordsService" as Srv
entity "WordsRepository" as Repo
database "MySQL" as DB
entity "CloudStorage" as Cloud

autonumber

U -> View: Accede a la sección de diccionario
View -> Api: getCategories()
Api -> Ctrl: GET /api/words/categories
Ctrl -> Srv: fetchAllCategories()
Srv -> Repo: findDistinctCategories()
note right: Las palabras se dividen en categorías\npara facilitar la búsqueda
Repo -> DB: SELECT category FROM word
DB --> Repo: categoriesList
Repo --> Srv: categoriesList
Srv --> Ctrl: categoriesList
Ctrl --> Api: 200 OK
Api --> View: Renderizar categorías

U -> View: Selecciona sección (palabras, frases, números)
View -> Api: getItems(section, page)
Api -> Ctrl: GET /api/words/{section}?page=...
Ctrl -> Srv: getPaginatedItems(section, page)
Srv -> Repo: findBySection(section, pageable)
Repo -> DB: SELECT * FROM word WHERE type=? LIMIT ...
DB --> Repo: itemsPage
Repo --> Srv: itemsPage
Srv --> Ctrl: itemsPage
Ctrl --> Api: 200 OK
Api --> View: Mostrar conjunto paginado

U -> View: Selecciona una entrada (palabra/frase)
View -> Api: getItemDetail(id)
Api -> Ctrl: GET /api/words/detail/{id}
Ctrl -> Srv: getFullDetail(id)

Srv -> Repo: findById(id)
Repo -> DB: SELECT esp_text, maz_text, audio_path, image_path FROM ...
DB --> Repo: itemDetails
Repo --> Srv: itemDetails

note over Srv: Procesamiento de recursos opcionales

alt #FFFDD0 Entrada contiene Audio
    Srv -> Cloud: getSignedUrl(audio_path)
    Cloud --> Srv: audioUrl
end

alt #FFFDD0 Entrada contiene Imagen
    Srv -> Cloud: getSignedUrl(image_path)
    Cloud --> Srv: imageUrl
end

Srv --> Ctrl: WordsDetailDto (Texts + Optional URLs)
Ctrl --> Api: 200 OK
Api --> View: RenderFullView
View --> U: Muestra traducción (obligatoria), imagen y audio (si existen)

@enduml
