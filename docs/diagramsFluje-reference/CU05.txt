@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-005 Registrar usuarios externos

actor "Usuario" as U
boundary "RegistroExternoView" as View
control "ApiService" as Api
control "ExternalAuthController" as Ctrl
entity "ExternalAuthService" as Srv
entity "UserRepository" as Repo
database "MySQL" as DB

autonumber

U -> View: Ingresa a la página de registro
View --> U: Muestra formulario de registro

U -> View: Completa correo, nombre y contraseña
View -> Api: registerExternalUser(dto)
Api -> Ctrl: POST /api/external/register (dto)
Ctrl -> Srv: createExternalAccount(dto)

alt #FFFDD0 Los campos están vacíos
    Srv --> Ctrl: throw ValidationException
    Ctrl --> Api: 400 Bad Request ("Campos vacíos")
    Api --> View: Error detectado
    View --> U: Mostrar: "Los campos están vacíos"
end

Srv -> Repo: findByEmail(email)
Repo -> DB: SELECT * FROM user WHERE email = ?
DB --> Repo: UserEntity / null
Repo --> Srv: data / null

alt #FFFDD0 El correo electrónico ya ha sido usado
    Srv --> Ctrl: throw DuplicateEmailException
    Ctrl --> Api: 409 Conflict ("Email en uso")
    Api --> View: Error detectado
    View --> U: Mostrar: "El correo electrónico ya ha sido usado"
end

alt #FFFDD0 La contraseña no cumple con los requisitos
    Srv -> Srv: checkPasswordStrength(password)
    Srv --> Ctrl: throw WeakPasswordException
    Ctrl --> Api: 400 Bad Request ("Contraseña débil")
    Api --> View: Error detectado
    View --> U: Mostrar: "La contraseña no cumple con los requisitos"
end

Srv -> Srv: encryptPassword(password)
note right: Los datos se almacenan con cifrado por seguridad

Srv -> Repo: save(UserEntity)
Repo -> DB: INSERT INTO user (visitor)
DB --> Repo: confirmation
Repo --> Srv: storedEntity

Srv --> Ctrl: SuccessResponse
Ctrl --> Api: 201 Created
Api --> View: SuccessNotification
View -> U: Redirigir a página de login como externo

@enduml
