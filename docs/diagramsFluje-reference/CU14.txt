@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-014: Asignar puntuaciones a actividades

actor "Docente" as T
boundary "ConfiguracionActividadView" as View
control "ApiService" as Api
control "ActivityController" as Ctrl
entity "ExperienceService" as Srv
entity "ActivityRepository" as Repo
database "MySQL" as DB

autonumber

T -> View: Ingresa al panel y selecciona actividad
View -> Api: getSuggestedScore(activityId)
Api -> Ctrl: GET /api/activities/{id}/suggested-score
Ctrl -> Srv: calculateRecommendedXP(activityId)

Srv -> Srv: Evaluar dificultad y cantidad de palabras
note right: El sistema calcula automáticamente la\nexperiencia recomendada

Srv --> Ctrl: suggestedXP
Ctrl --> Api: 200 OK (suggestedXP)
Api --> View: Mostrar puntuación recomendada

T -> View: Confirma o modifica el valor de experiencia
View -> Api: updateActivityScore(activityId, customXP)
Api -> Ctrl: PATCH /api/activities/{id}/score

group Validación de Rango (Backend)
    Ctrl -> Srv: validateAndApplyXP(activityId, customXP)
    
    alt #FFFDD0 Valor fuera de rango o vacío
        Srv --> Ctrl: throw InvalidXPException
        Ctrl --> Api: 422 Unprocessable Entity
        Api --> View: Error detectado (Rango excedido)
        View --> T: Alerta: "Valor fuera de rango, se usará el valor por defecto"
    end
end

Srv -> Repo: updateXP(activityId, validatedXP)
Repo -> DB: UPDATE activity SET experience = ? WHERE id = ?
DB --> Repo: confirmation
Repo --> Srv: success

Srv --> Ctrl: updatedScoreDto
Ctrl --> Api: 200 OK
Api --> View: notifySuccess
View --> T: Notifica asignación exitosa

@enduml
