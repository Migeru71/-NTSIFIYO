@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-011: Recopilar tiempo de uso de la aplicación

actor "Usuario" as U
boundary "Vista" as View
control "ApiService" as Api
control "SessionController" as Ctrl
entity "SessionService" as Srv
entity "SessionRepository" as Repo
database "MySQL" as DB

autonumber

U -> View: Entra a la página o inicia aplicación
View -> Api: startTracking(userId)

alt #FFFDD0 No hay conexión a internet
    Api -> Api: Almacenar momento de inicio en LocalStorage
end

Api -> Ctrl: POST /api/sessions/start (userId)
Ctrl -> Srv: beginSession(userId)

Srv -> Srv: Identificar usuario (ID o Anónimo)

Srv -> Repo: save(SessionEntity)
Repo -> DB: INSERT INTO sesion (user_id, start_date)
DB --> Repo: confirmation
Repo --> Srv: SessionEntity
Srv --> Ctrl: sessionId
Ctrl --> Api: 200 OK (sessionId)

U -> View: Cierra la pestaña o aplicación
View -> Api: stopTracking(sessionId)

alt #FFFDD0 No hay conexión a internet
    Api -> Api: Almacenar momento de finalización localmente
end

Api -> Ctrl: PUT /api/sessions/end (sessionId)
Ctrl -> Srv: finishSession(sessionId)

Srv -> Srv: Calcular duración total

Srv -> Repo: update(SessionEntity)
Repo -> DB: UPDATE sesion SET duration = ? WHERE id = ?
DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: success
Ctrl --> Api: 200 OK

@enduml
