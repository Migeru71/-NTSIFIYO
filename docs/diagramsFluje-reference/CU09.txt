@startuml
skinparam sequence {
    LifeLineBackgroundColor #F5F5F5
    ParticipantBackgroundColor #FFFDD0
    ParticipantBorderColor #666666
}
skinparam NoteBackgroundColor #F5F5F5

title CU-009: Crear grupos

actor "Administrador" as Admin
boundary "CrearGrupoView" as View
control "ApiService" as Api
control "GroupController" as Ctrl
entity "GroupService" as Srv
entity "UserRepository" as UserRepo
entity "GroupRepository" as Repo
database "MySQL" as DB

autonumber

Admin -> View: Accede a la sección de grupos
View -> Api: fetchCreationStatus()
Api -> Ctrl: GET /api/groups/creation-status
Ctrl -> Srv: checkCreationEligibility()

group Validación de Límites (Pre-creación)
    Srv -> Repo: countGroups()
    Repo -> DB: SELECT COUNT(*) FROM group
    DB --> Repo: count
    Repo --> Srv: count
    
    alt #FFFDD0 Ya hay 6 grupos creados (Límite de primaria)
        Srv --> Ctrl: throw LimitReachedException
        Ctrl --> Api: 403 Forbidden ("Límite alcanzado")
        Api --> View: Opción deshabilitada
        View --> Admin: Mostrar: "No se pueden crear más grupos (Límite: 6)"
    end
end

Srv --> Ctrl: EligibleGradesList
Ctrl --> Api: 200 OK
Api --> View: RenderForm (Grados disponibles)

Admin -> View: Selecciona grado y solicita listas de asignación
View -> Api: fetchAvailableStaffAndStudents()
Api -> Ctrl: GET /api/users/available-for-group
Ctrl -> Srv: getCandidates()

Srv -> UserRepo: findUnassignedStudentsAndTeachers()
note right: Los alumnos ya inscritos en otros grupos\nno se mostrarán en el listado
UserRepo -> DB: SELECT * FROM user WHERE ...
DB --> UserRepo: candidatesList
UserRepo --> Srv: candidatesList
Srv --> Ctrl: candidatesList
Ctrl --> Api: 200 OK
Api --> View: ShowSelectionLists

Admin -> View: Asigna docente, alumnos y confirma creación
View -> Api: createGroup(groupDto)
Api -> Ctrl: POST /api/groups
Ctrl -> Srv: processGroupCreation(dto)

alt #FFFDD0 Intento de crear grado ya existente
    Srv -> Repo: existsByGrade(grade)
    Repo -> DB: SELECT id FROM group WHERE grade = ?
    DB --> Repo: id / null
    
    alt #FFFDD0 Grado duplicado
        Srv --> Ctrl: throw DuplicateGradeException
        Ctrl --> Api: 409 Conflict
        Api --> View: Error detectado
        View --> Admin: Mostrar: "Ya existe un grupo con ese grado"
    end
end

alt #FFFDD0 Falta docente o alumnos
    Srv -> Srv: validateMinimumStaff(dto)
    alt #FFFDD0 Sin docente o sin alumnos
        Srv --> Ctrl: throw IncompleteAssignmentException
        Ctrl --> Api: 400 Bad Request
        Api --> View: Error detectado
        View --> Admin: Mostrar: "Se requiere un docente y al menos un alumno"
    end
end

Srv -> Repo: save(GroupEntity)
Repo -> DB: INSERT INTO group (...)
DB --> Repo: success
Repo --> Srv: confirmation
Srv --> Ctrl: SuccessDto
Ctrl --> Api: 201 Created
Api --> View: SuccessNotification
View --> Admin: Notifica creación exitosa

@enduml
