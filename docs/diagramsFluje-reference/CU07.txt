control "RecommendationController" as Ctrl
entity "RecommendationService" as Srv

autonumber

U -> View: Ingresa a la página principal
View -> View: Generar fecha actual (Filtro temporada)
View -> Api: getRecommendations(currentDate)
Api -> Ctrl: GET /api/recommendations?date=...
Ctrl -> Srv: getPersonalizedContent(date)

Srv -> Srv: Extraer UserId de la sesión (JWT)
Srv -> Repo: findUserDtoById(userId)
Repo -> DB: SELECT grade, level FROM user WHERE id = ?
DB --> Repo: UserDto (grade, level)
Repo --> Srv: UserDto

group Proceso de Recomendación
    Srv -> Repo: findByGradeExpAndSeason(grade, diff, date)
    Repo -> DB: SELECT * FROM media WHERE ...
    DB --> Repo: seasonalContentList
    Repo --> Srv: seasonalContentList

    alt #FFFDD0 No hay contenido para esa temporada del año
        Srv -> Repo: findByGradeAndExp(grade, diff)
        Repo -> DB: SELECT * FROM media WHERE grade=? AND diff=?
        DB --> Repo: fallbackContentList
        Repo --> Srv: fallbackContentList
        note right: Se salta el filtrado por temporada
    end
end

Srv --> Ctrl: RecommendedContentDto
Ctrl --> Api: 200 OK
Api --> View: RenderRecommendedSection
View --> U: Muestra contenido en sección de recomendados
